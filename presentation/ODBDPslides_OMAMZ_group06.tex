        \item 70\% chance to be chosen
      \end{itemize}
      \item Inversion
      \begin{itemize}
        \item traditional approach adapted to the instance dimension
        \item 15\% chance to be chosen
      \end{itemize}
      \item Crossover
       \begin{itemize}
        \item traditional approach
        \item 15\% chance to be chosen
      \end{itemize}
